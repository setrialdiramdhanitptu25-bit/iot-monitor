#include #include #include "DHT.h" // ======= Konfigurasi WiFi ======= const char* ssid = "POCO X5 5G"; const char* password = "123456789"; // ======= Konfigurasi Web App ======= const char* scriptURL = "https://script.google.com/macros/s/AKfycbxIkWUVLapdydhv6nLPeTq1E2DHeBupOmL1zxMvJAt24G3y1EbsK3K_e9FEMUFFTt3qBw/exec"; // ======= Konfigurasi Sensor ======= #define DHTPIN 4 #define DHTTYPE DHT11 DHT dht(DHTPIN, DHTTYPE); #define MQ2PIN 34 // Interval kirim data (ms) unsigned long interval = 5000; // 5 detik unsigned long previousMillis = 0; void setup() { Serial.begin(115200); dht.begin(); pinMode(MQ2PIN, INPUT); Serial.print("Connecting to WiFi"); WiFi.begin(ssid, password); while(WiFi.status() != WL_CONNECTED){ delay(500); Serial.print("."); } Serial.println("\nWiFi Connected"); } void loop() { unsigned long currentMillis = millis(); if(currentMillis - previousMillis >= interval){ previousMillis = currentMillis; // ===== Baca Sensor ===== float temp = dht.readTemperature(); float hum = dht.readHumidity(); int gas = analogRead(MQ2PIN); if (isnan(temp) || isnan(hum)) { Serial.println("Failed to read DHT sensor!"); return; } Serial.println("Temperature: " + String(temp) + "°C"); Serial.println("Humidity: " + String(hum) + "%"); Serial.println("Gas: " + String(gas)); // ===== Kirim ke Spreadsheet ===== if(WiFi.status() == WL_CONNECTED){ HTTPClient http; String url = String(scriptURL) + "?temperature=" + temp + "&humidity=" + hum + "&gas=" + gas; http.begin(url); int httpCode = http.GET(); if(httpCode > 0){ Serial.println("Data sent, response: " + String(httpCode)); } else { Serial.println("Error sending data"); } http.end(); } else { Serial.println("WiFi not connected"); } } }
